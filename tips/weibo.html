n_features = 1000 #жҸҗеҸ–1000дёӘзү№еҫҒиҜҚиҜӯ

                                max_features=n_features,

иҝҷдёҖдёІд»Јз Ғдёӯпјҡ

                                learning_method='batch',
                                learning_offset=50,
                                 doc_topic_prior=0.1,
                                 topic_word_prior=0.01,
                               random_state=0)
lda.fit(tf)

иҝҷдёҖдёІд»Јз Ғдёӯпјҡ

  • n_topicsзҡ„ж•°еҖјеҸҜд»Ҙж №жҚ®йңҖиҰҒиҮӘиЎҢдҝ®ж”№гҖӮ
